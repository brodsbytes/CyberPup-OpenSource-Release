9. DYNAMIC CHECKLIST GENERATION
===============================

Purpose: Create device-specific checklist items with individual completion tracking
Location: utils/checklistGenerator.js

Features:
• Generate separate checklist items for each user device
• Track completion per device for gratification
• Aggregate completion for overall check progress
• Handle mixed device scenarios intelligently

Implementation:
```javascript
export class ChecklistGenerator {
  static generateForDevices(baseChecklist, userDevices, checkId) {
    // Determine if check is device-specific or universal
    const checkType = this.getCheckType(checkId);
    
    if (checkType === 'universal') {
      // Same checklist for all devices (e.g., password strength)
      return this.generateUniversalChecklist(baseChecklist);
    } else if (checkType === 'device-specific') {
      // Separate checklist per device (e.g., screen lock settings)
      return this.generateDeviceSpecificChecklists(baseChecklist, userDevices);
    } else if (checkType === 'mixed') {
      // Some universal, some device-specific items
      return this.generateMixedChecklist(baseChecklist, userDevices);
    }
  }

  static generateDeviceSpecificChecklists(baseChecklist, userDevices) {
    const deviceChecklists = [];
    
    userDevices.forEach(device => {
      const deviceItems = this.adaptChecklistForDevice(baseChecklist, device);
      deviceChecklists.push({
        deviceId: device.id,
        deviceName: device.name,
        deviceType: device.type,
        items: deviceItems,
        completed: false,
        progress: 0
      });
    });
    
    return deviceChecklists;
  }

  static adaptChecklistForDevice(checklist, device) {
    return checklist.map(item => ({
      ...item,
      text: this.customizeTextForDevice(item.text, device),
      action: this.getDeviceSpecificAction(item.action, device),
      helpText: this.getDeviceSpecificHelp(item.helpText, device)
    }));
  }
}
```

    // Show success feedback or guidance for improvement
    if (isValid) {
      this.showSuccessFeedback();
    } else {
      this.showImprovementGuidance();
    }
    
    onComplete(item.id, isValid);
  };

  const validateAction = async (itemId, device) => {
    // For items that can be programmatically verified
    switch (itemId) {
      case 'screen-lock-enabled':
        return await this.checkScreenLockStatus(device);
      case 'auto-updates-enabled':
        return await this.checkUpdateSettings(device);
      default:
        return true; // Self-reported items
    }
  };
};

// Storage utility extensions (add to utils/storage.js)
export const EnhancedStorageUtils = {
  // Device-specific progress
  async getDeviceProgress(checkId, deviceId) {
    const key = `${ENHANCED_STORAGE_KEYS.DEVICE_PROGRESS_PREFIX}${checkId}_${deviceId}`;
    return await StorageUtils.getObject(key);
  },
  
  async setDeviceProgress(checkId, deviceId, progress) {
    const key = `${ENHANCED_STORAGE_KEYS.DEVICE_PROGRESS_PREFIX}${checkId}_${deviceId}`;
    return await StorageUtils.setObject(key, progress);
  },
  
  // Practice results
  async getPracticeResults(checkId) {
    const key = `${ENHANCED_STORAGE_KEYS.PRACTICE_RESULTS_PREFIX}${checkId}`;
    return await StorageUtils.getObject(key);
  },
  
  async savePracticeResult(checkId, result) {
    const key = `${ENHANCED_STORAGE_KEYS.PRACTICE_RESULTS_PREFIX}${checkId}`;
    const existing = await this.getPracticeResults(checkId) || [];
    existing.push(result);
    return await StorageUtils.setObject(key, existing);
  }
